161static inline uint16_t libusb_cpu_to_le16(const uint16_t x)
162{
163 union {
164 uint8_t b8[2];
165 uint16_t b16;
166 } _tmp;
167 _tmp.b8[1] = (uint8_t) (x >> 8);
168 _tmp.b8[0] = (uint8_t) (x & 0xff);
169 return _tmp.b16;
170}
